\subsection[Construction of K\_pi]{Construction of $K_\pi$ (continued)}

  \begin{remark}
    Recall that $[\pi]_f(T) = f(T)$.
    And therefore 
    \[ \Lambda_n = \text{ roots of } f^{\circ n} \text{ in } K^\al \text{ with } |\cdot | < 1. \]
  \end{remark}

  For simplicity, assume 
  \[ f = \pi T + a_2 T^2 + \cdots + T^q. \]
  (Even $\pi T + T^q$ is enough.)
  By induction, 
  \[ f^{\circ n} = \pi^n T + \text{ middle terms } + T^{q^n}. \]
  Here the middle terms are divisible by $\pi$.
  From Newton polygon of $f^{\circ n}$, all roots of $f^{\circ n}$ have norm less than $1$.
  Hence $\Lambda_n$ is the set of roots of $f^{\circ n}$ in $K^\al$.
  On $\Lambda_n$, we endow the commutative group structure given by $F_f$ and the $A$-module structure given by $[a]_f$.

  \begin{example}
    Take $K = \bbQ_p$ and $f = (T+1)^p - 1 \in \calF_p$.
    Then 
    \[ \Lambda_n = \{ \alpha \in \bbQ_p^\al : (\alpha - 1)^{p^n} = 1 \} \cong \mu_{p^n}(\bbQ_p^\al). \]
    The addition $+_F$ on $\Lambda_f$ corresponds to the multiplication of $\mu_{p^n}(\bbQ_p^\al)$, and the $A$-module structure corresponds to the Galois action of $\Gal(\bbQ_p(\mu_{p^n})/\bbQ_p)$.
    It follows that $\Lambda_n \cong \bbZ_p / p^n \bbZ_p$ as $\bbZ_p$-modules.
  \end{example}

  Recall the \emph{cyclic module} is a module generated by one element.
  Since in our case $A$ is a DVR, every finitely generated torsion $A$-module has a unique decomposition into cyclic modules:
  \[ M \cong A / \pi^{n_1} A \oplus \cdots \oplus A / \pi^{n_r} A, \]
  where $n_1 \leq \cdots \leq n_r$.

  \begin{lemma}\label{lem:criterion_of_cyclic_module}
    Let $M$ be a $A$-module.
    Let $M_n = \Ker (\pi^n : M \to M)$.
    Assume $M_1$ has $q = \#(A/\pi A)$ elements and $\pi: M \to M$ is surjective.
    Then $M_n \cong A / \pi^n A$ for all $n$.
    In particular, $\# M_n = q^n$ for all $n$.
  \end{lemma}
  \begin{proof}
    Induction on $n$.
    As $\# A / \pi A = q$, we have $M_1 \cong A / \pi A$ by the structure theorem of modules over a PID.
    Consider the exact sequence
    \[ 0 \to M_1 \to M_n \xrightarrow{\pi} M_{n-1} \to 0. \]
    Hence $\# M_n = \# M_1 \cdot \# M_{n-1} = q^n$.
    If $M_n$ is not cyclic, then $\Ker (\pi : M_n \to M_{n-1})$ has at least $q^2$ elements, contradicting the fact that $\Ker (\pi : M_n \to M_{n-1}) = M_1$ has $q$ elements.
  \end{proof}

  \begin{proposition}
    The $A$-module $\Lambda_n$ is isomorphic to $A / \pi^n A$.
    Hence $\End_A(\Lambda_n) \cong A / \pi^n A$ and $\Aut_A(\Lambda_n) \cong (A / \pi^n A)^\times$.
  \end{proposition}
  \begin{proof}
    An $A$-isomorphism $h: F_f \to F_g$ of formal group law 
    \Yang{}
    By \cref{lem:criterion_of_cyclic_module}, we get $\Lambda_n \cong A / \pi^n A$.
    And the other two isomorphisms follow immediately.
  \end{proof}

  \begin{lemma}\label{lem:Galois_action_on_power_series_over_complete_field}
    Let $L/K$ be a finite Galois extension with Galois group $G$.
    For all $F \in \calO_K[[X_1, \ldots, X_n]]$ and $\alpha_1, \ldots, \alpha_n \in \frakm_L$, we have
    \[ \tau(F(\alpha_1, \ldots, \alpha_n)) = F(\tau(\alpha_1), \ldots, \tau(\alpha_n)) \]
    for all $\tau \in G$.
  \end{lemma}
  \begin{proof}
    Use polynomial approximation of $F$ and the continuity of $\tau$.
  \end{proof}

  \begin{remark}
    If $K$ is not complete, the above lemma may fail.
    For example, take $K = \bbQ$ and $L = \bbQ(\sqrt{2})$, and $F = \exp$.
  \end{remark}

  \begin{theorem}
    Let $K_{\pi, n} := K(\Lambda_n)$.
    \begin{enumerate}
      \item $\forall n$, $K_{\pi, n} / K$ is totally ramified of degree $q^{n-1}(q-1)$.
      \item The action of $A$ on $\Lambda_n$ defines an isomorphism $A / \pi^n A \cong \Gal(K_{\pi, n} / K)$. 
        In particular, $K_{\pi, n} / K$ is an abelian extension.
      \item $\forall n$, $\pi$ is a norm from $K_{\pi, n}$ to $K$.
    \end{enumerate}
  \end{theorem}
  \begin{proof}
    We may assume $f = \pi T + \cdots + T^q$.

    \paragraph{(a) and (b)} 
    Choose a nonzero root $\pi_1$ of $f(T)$ and (inductively) a nonzero root $\pi_n$ of $f(T) - \pi_{n-1}$ for all $n$.
    Then $\pi_n$ is a root of $f^{\circ n}(T)$ and hence $\pi_n \in \Lambda_n$.
    Consider a sequence of fields
    \[ K(\Lambda_n) \supset K(\pi_n) \supset K(\pi_{n-1}) \supset \cdots \supset K(\pi_1) \supset K. \]
    We have $K(\pi_1) / K$ is totally ramified of degree $q-1$ by Eisenstein's criterion.
    
    We claim that $K(\pi_n) / K$ is totally ramified of degree $q^{n-1}(q-1)$ and $|\pi_n| = |\pi|^{1/(q^{n-1}(q-1))}$.
    We prove the claim by induction on $n$.
    The case $n=1$ is already proved.
    Note that $\pi_{n+1}$ is a root of
    \[ f(T) - \pi_n = T^q + \text{ middle terms } + \pi T - \pi_n, \]
    which is an Eisenstein polynomial over $K(\pi_n)$ by the induction hypothesis.
    Hence $K(\pi_{n+1}) / K(\pi_n)$ is totally ramified of degree $q$ and $|\pi_{n+1}| = |\pi_n|^{1/q}$.

    Note that $K(\Lambda_n)$ is the splitting field of $f^{\circ n}(T)$, which is a separable polynomial.
    Hence $K(\Lambda_n) / K$ is Galois.
    Then $\Gal(K(\Lambda_n) / K)$ acts on $\Lambda_n$ and preserves the $A$-module structure by \cref{lem:Galois_action_on_power_series_over_complete_field}.
    Hence we get a homomorphism $\Gal(K(\Lambda_n) / K) \injmap \Aut_A(\Lambda_n) \cong (A / \pi^n A)^\times$.
    Counting the cardinality, we get
    \[ q^{n-1}(q-1) = [K(\pi_n) : K] \leq \# \Gal(K(\Lambda_n) / K) \leq \# \Aut_A(\Lambda_n) = q^{n-1}(q-1). \]
    Hence all inequalities are equalities, and the homomorphism is an isomorphism.

    \paragraph{(c)}
    Set 
    \[ f^{[n]}(T) := (f/T) \circ f \circ \cdots \circ f(T) = \pi + \text{ middle terms } + T^{q^n - 1}. \]
    Then $f^{[n]}(\pi_n) = f^{[n-1]}(\pi_{n-1}) = \cdots = f^{[1]}(\pi_1) = 0$.
    As $f^{[n]}(T)$ is an Eisenstein polynomial and of degree $(q-1)q^{n-1}$, it is the minimal polynomial of $\pi_n$ over $K$.
    Hence $\pi$ is the norm of $\pi_n$ from $K(\pi_n)$ to $K$.
  \end{proof}

  In summary, let $f = \pi T + \cdots + T^q$ and $\Lambda_n$ be the set of roots of $f^{\circ n}$ in $K^\al$ and $K_{\pi, n} = K(\Lambda_n)$.
  Then $K_{\pi, n} = K(\pi_n)$ with $f(\pi_n) = \pi_{n-1}$, and $\frakm_{K_{\pi, n}} = \pi_n \calO_{K_{\pi, n}}$.
  Moreover, $a.\lambda = [a]_f(\lambda)$ for all $a \in A$ and $\lambda \in \Lambda_n$ induces an isomorphism $(A / \pi^n A)^\times \cong \Gal(K_{\pi, n} / K)$.

  Take the inverse limit of $\Lambda_n$ and $\Gal(K_{\pi, n} / K)$, we get an isomorphism
  \[ A^\times \cong \Gal(K_\pi / K), \]
  where $K_\pi := \bigcup_{n=1}^\infty K_{\pi, n}$.

  \begin{example}
    Let $K = \bbQ_p$ and $f = (T+1)^p - 1$.
    For all $r$, choose a primitive $p^r$-th root of unity $\zeta_{p^r}$ such that $\zeta_{p^{r+1}}^p = \zeta_{p^r}$.
    Then $\pi_r := \zeta_{p^r} - 1$.
    Then 
    \[ (\bbQ_p)_{p,r} = \bbQ_p(\pi_r) = \bbQ_p(\zeta_{p^r}). \]
    Moreover, the isomorphism $(\bbZ_p / p^r \bbZ_p)^\times \cong \Gal(\bbQ_p(\zeta_{p^r}) / \bbQ_p)$ is given by $a + p^r \bbZ_p \mapsto (\zeta_{p^r} \mapsto \zeta_{p^r}^a)$.
  \end{example}


\subsection{The local Artin map}

  Recall that $M,N/K$ are Galois extensions such that $M \cap N = K$.
  Then we have an isomorphism $\Gal(MN/K) \cong \Gal(M/K) \times \Gal(N/K)$.
  
  We define 
  \begin{align*}
    \phi_\pi : & K^\times \to \Gal(K_\pi K^\unr / K) \cong \Gal(K_\pi / K) \times \Gal(K^\unr / K) \\
    & a = \pi^m u \mapsto ([u^{-1}]_f, \Frob_K^m).
  \end{align*} 
  Here $\Frob_K$ is the Frobenius element of $\Gal(K^\unr / K)$ and $[u^{-1}]_f$ is the automorphism of $K_\pi$ induced by the $A$-module structure on $\Lambda_n$.

  \begin{theorem}
    Both $K_\pi \cdot K^\unr$ and $\phi_\pi$ are independent of the choice of $\pi$.
  \end{theorem}

  Set $\widehat{K^\unr}$ be the completion of $K^\unr$ with valuation ring $B$.
  Then $\Frob_K$ extends to $\sigma \in \Gal(\widehat{K^\unr} / K)$.
  For a power series $\theta(T) = \sum_i b_i T^i \in B[[T]]$, we define $\sigma(\theta)(T) = \sum_i (\sigma(b_i))T^i$.

  \begin{proposition}
    Let $\varpi = u \pi$ with $\pi$ a uniformizer and $u \in A^\times$.
    Let $f \in \calF_\pi$ and $g \in \calF_\varpi$.
    Then there exists $\varepsilon \in B^\times$ such that $\sigma(\varepsilon) = \varepsilon u$ and a power series $\theta(T) \in T B[[T]]$ such that
    \begin{enumerate}
      \item $\theta(T) \equiv \varepsilon T \mod T^2$;
      \item $\sigma(\theta) = \theta \circ [u]_f$;
      \item $\theta(F_f(X,Y)) = F_g(\theta(X), \theta(Y))$;
      \item $\theta \circ [a]_f = [a]_g \circ \theta$ for all $a \in A$.
    \end{enumerate}
  \end{proposition}
