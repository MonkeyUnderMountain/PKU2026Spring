  \begin{lemma}\label{lem:surjective_of_sigma_minus_id}
    The homomorphisms $B \to B, b \mapsto \sigma(b) - b$ and $B^\times \to B^\times, b \mapsto \sigma(b)/b$ are surjective with kernels $A$ and $A^\times$ respectively.
  \end{lemma}
  \begin{proof}
    Let $R$ be the valuation ring of $K^\unr$ and $\frakn$ its maximal ideal.
    Then $R$ is a DVR and $B = \invlim_n R/\frakn^n$.
    
    Show by induction that 
    \[ 0 \to A/\frakm^n \to R/\frakn^n \xrightarrow{\sigma - \id} R/\frakn^n \to 0 \]
    is exact for all $n$.
    For $n = 1$, 
    \[ 0 \to k \to \overline{k} \xrightarrow{\sigma - \id} \overline{k} \to 0 \]
    is exact since $\overline{k}$ is algebraically closed.
    Assume the exactness for $n-1$.
    Consider
    \[ \begin{tikzcd}
      0 \arrow[r] & R/\frakn \arrow[r] \arrow[d, "\sigma - \id"] & R/\frakn^{n} \arrow[r] \arrow[d, "\sigma - \id"] & R/\frakn^{n-1} \arrow[r] \arrow[d, "\sigma - \id"] & 0 \\
      0 \arrow[r] & R/\frakn \arrow[r] & R/\frakn^n \arrow[r] & R/\frakn^{n-1} \arrow[r] & 0
    \end{tikzcd} \]
    By the snake lemma, we have an exact sequence
    \[ 0 \to A/\frakm \to \ker(\sigma - \id)  \to A/\frakm^{n-1} \to 0 \to \coker(\sigma - \id) \to 0. \]
    Hence $\sigma - \id$ is surjective and $\# \ker(\sigma - \id) = q \times q^{n-1} = q^n$.
    Note that $A/\frakm^n \subset \ker(\sigma - \id)$ and $\# A/\frakm^n = q^n$.
    Hence $\ker(\sigma - \id) = A/\frakm^n$.
    This proves the exactness for $n$.
    Take the inverse limit, we get the exactness of $0 \to A \to B \xrightarrow{\sigma - \id} B \to 0$.
    This proves the surjectivity of $B \to B, b \mapsto \sigma(b) - b$ with kernel $A$.

    The other exact sequence can be proved similarly.
    % \Yang{}
  \end{proof}

\subsection{Local Kronecker-Weber theorem}

  \begin{theorem}[Local Kronecker-Weber]
    We have $K^\ab = K^\ur K_\pi$ for any uniformizer $\pi$ of $A$.
  \end{theorem}

  The ramification groups of $K_{\pi,n}/K$.
  Let $L/K$ be a finite Galois extension and $G = \Gal(L/K)$.
  Recall the $i$-th ramification group $G_i$ of $L/K$ is defined as follows:
  \[ G_i = \{ \tau \in G : \ord_L(\tau(a) - a) \ge i + 1 \text{ for all } a \in \calO_L \}. \]
  Then $G_0$ is the inertia subgroup of $G$.
  Moreover, $G_i = \{ \tau \in G : \ord_L(\tau(\pi_L) - \pi_L) \ge i + 1 \}$ for any uniformizer $\pi_L$ of $L$.
  Note that $G/G_0$ is isomorphic to the Galois group of the residue field extension of $L/K$.
  There exists inclusion 
  \[ G_0/G_1 \hookrightarrow \call^\times, \pi \mapsto \tau(\pi)/\pi \mod \frakm_L. \]
  \[ G_i/G_{i+1} \hookrightarrow \call, \tau \mapsto \tau(\pi) - \pi \mod \frakm_L^{i+1}. \]
  Hence we have 
  \[ [G_0 : G_1] \mid q - 1, \quad [G_i : G_{i+1}] \mid q \text{ for } i \ge 1. \]
  Moreover, $G_i = \{ 1 \}$ for $i$ sufficiently large.
