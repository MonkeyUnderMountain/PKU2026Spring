  \begin{theorem}[Martingale Convergence Theorem]
    Let $(X, \calF, \mu)$ be a probability space and $T: X \to X$ be a measure-preserving transformation. 
    Let $\calA$ be a countable measurable partition of $X$ with finite entropy. 
    Then the sequence of functions $(I_{\calA | \calA^n})_{n \geq 0}$ converges a.e. and in $L^1(X, \calF, \mu)$ as $n \to \infty$. 
     \Yang{}
    
  \end{theorem}
