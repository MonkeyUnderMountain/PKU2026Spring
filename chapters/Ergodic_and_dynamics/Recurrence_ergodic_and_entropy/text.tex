\section{Recurrence, ergodic and entropy}


  We fix a probability space $(X, \calF, \mu)$.


\subsection{Recurrence}

  % Recall that for two measure spaces $(X, \calF, \mu)$ and $(Y, \calG, \nu)$, a measurable map $f: X \to Y$ is called \emph{measure-preserving} if for every $A \in \calG$, we have $\mu(f^{-1}(A)) = \nu(A)$.

  \begin{definition}
    A measurable transformation $T: X \to X$ is called \emph{measure-preserving} if for every $A \in \calF$, we have $\mu(T^{-1}(A)) = \mu(A)$.
  \end{definition}

  It is worth noting that $T$ is measure-preserving if and only if $\int_X f \circ T \, d\mu = \int_X f \, d\mu$ for every $f \in L^1(X, \calF, \mu)$.

  \begin{example}
    Consider the unit circle $S^1$ with the Lebesgue measure $\mu$. 
    Then the rotation $R_s: S^1 \to S^1, \exp(2\pi i t) \mapsto \exp(2\pi i (t + s))$ is measure-preserving for every $s \in \bbR$.
  \end{example}

  \begin{example}
    The power map $S^1 \to S^1, z \mapsto z^n$ is measure-preserving with respect to the Lebesgue measure on the unit circle.

    More generally, consider $\bbP^1(\bbC) \to \bbP^1(\bbC), z \mapsto z^n$ and $\mu$ is given by $i_* \frac{1}{2\pi} d\theta$ where $i: S^1 \to \bbP^1(\bbC)$ is the inclusion map. 
    Then this map is also measure-preserving.
  \end{example}


  \begin{theorem}[Poincar\'e Recurrence Theorem]
    Let $T: X \to X$ be a measure-preserving transformation. 
    For every measurable set $A$, we have 
    \[
      \mu\left(\{x \in A : T^n x \in A \text{ for infinitely many } n \in \mathbb{N}\}\right) = \mu(A).
    \]
    In other words, for almost every point in $A$, its orbit under $T$ returns to $A$ infinitely often.
    % \Yang{}
  \end{theorem}
  \begin{proof}
    Consider the set
    \[ N := \{x \in A : T^n x \notin A , \ \forall n \geq 1\}. \]
    Note that 
    \[ N = A \setminus \bigcup_{n=1}^\infty T^{-n}(A). \]
    Hence $N$ is a measurable set.
    If $x \in T^{-n}(N) \cap T^{-n-k}(N)$ for some $n, k \geq 1$, then $y = T^n x \in N$ and $T^k y = T^{n+k} x \in N \subseteq A$, which contradicts the definition of $N$.
    Hence the sets $T^{-n}(N)$ for $n \geq 1$ are pairwise disjoint.
    Then we have 
    \[ \sum_{n=1}^\infty \mu(T^{-n}(N)) = \sum_{n=1}^\infty \mu(N) \leq \mu(X) = 1. \]
    Hence $\mu(N) = 0$.

    Let $N_k = \{x \in A : T^{nk} x \notin A , \ \forall n \geq 1\}$ for $k \geq 1$. 
    Then $N_k$ is a measurable set and $\mu(N_k) = 0$ by the same argument as above.
    And we have
    \[ \{x \in A : T^n x \in A \text{ for finitely many } n \in \mathbb{N}\} = \bigcup_{k=1}^\infty N_k. \]
    Hence $\mu\left(\{x \in A : T^n x \in A \text{ for finitely many } n \in \mathbb{N}\}\right) = 0$.
    Therefore, we have
    \[
      \mu\left(\{x \in A : T^n x \in A \text{ for infinitely many } n \in \mathbb{N}\}\right) = \mu(A).
    \]
    We are done.
    % \Yang{}
  \end{proof}

\subsection{Ergodic}

  \begin{definition}[Conditional expectation]
    Let $(X, \calF, \mu)$ be a probability space and $\calG$ be a sub-$\sigma$-algebra of $\calF$. 
    For every $f \in L^1(X, \calF, \mu)$, there exists a unique $\calG$-measurable function $g$ such that for every $A \in \calG$, we have
    \[ \int_A g \, d\mu = \int_A f \, d\mu. \]
    We denote this function $g$ by $\bbE(f | \calG)$ or $f_\calG$ and call it the \emph{conditional expectation} of $f$ given $\calG$.
    % \Yang{}
  \end{definition}
  \begin{remark}
    The existence and uniqueness of $\bbE(f | \calG)$ can be proved by the Radon-Nikodym theorem.
  \end{remark}

  \begin{example}
    Let $\calG = \sigma(\calA)$ be the $\sigma$-algebra generated by a finite measurable partition $\calA$ of $X$. 
    Then for every $f \in L^1(X, \calF, \mu)$, we have
    \[ \bbE(f | \calG)(x) = \frac{1}{\mu(A_x)} \int_{A_x} f \, d\mu, \]
    where $A_x$ is the unique element of $\calA$ containing $x$.
  \end{example}

  \begin{example}
    Let $\calB$ be a measurable partition of $X$ and $\calG = \sigma(\calB)$ be the $\sigma$-algebra generated by $\calB$. 
    Consider the function $\phi(x) := \mu(A|B_x)$ where $A$ is a fixed measurable set and $B_x$ is the unique element of $\calB$ containing $x$. 
    Then $\phi$ is $\calG$-measurable and for every $C = \bigcup_i B_i \in \calG$, we have
    \[ \int_C \phi \, d\mu = \int_C \mu(A|B_x) \, d\mu = \sum_i \int_{B_i} \mu(A|B_x) \, d\mu = \sum_i \mu(A \cap B_i) = \mu(A \cap C) = \int_C \bbone_A \, d\mu. \]
    Hence we have $\bbE(\bbone_A | \calG)(x) = \mu(A|B_x)$.
  \end{example}

  % The conditional expectation is related to the conditional measure by 
  % \[ \mu(A | \calG)(x) = \bbE(\chi_A | \calG)(x) \]
  % for every $A \in \calF$ and $x \in X$.

  \begin{theorem}[Martingale Convergence Theorem]
    Let $(X, \calF, \mu)$ be a probability space and $\calG_n$ be an increasing sequence of sub-$\sigma$-algebras of $\calF$. 
    For every $f \in L^1(X, \calF, \mu)$, the sequence $\bbE(f | \calG_n)$ converges a.e. and in $L^1(X, \calF, \mu)$ as $n \to \infty$.
  \end{theorem}
  % \begin{example}
  % \end{example}

  \begin{definition}
    A measurable transformation $T: X \to X$ is called \emph{ergodic} if for every $A \in \calF$ such that $T^{-1}(A) \aeeq A$, we have $\mu(A) = 0$ or $\mu(A) = 1$. 
  \end{definition}

  In the definition, we do not require $\mu$ is $T$-invariant.

  Recall that our $X$ is a probability space.
  In other words, the only $T$-invariant sets are trivial (i.e., empty set and the whole space up to measure zero).

  Let $T: X \to X$ be a measure-preserving transformation and $\calI = \{A \in \calF : T^{-1}(A) \aeeq A\}$ be the $\sigma$-algebra of $T$-invariant sets. 
  Then for every $f \in L^1(X, \calF, \mu)$, we have $f$ is $\calI$-measurable if and only if $f \circ T \aeeq f$.

  \begin{example}
    Let $s$ be an irrational number. 
    Consider the rotation $R_s: S^1 \to S^1, \exp(2\pi i t) \mapsto \exp(2\pi i (t + s))$.
    Then $R_s$ is ergodic with respect to the Lebesgue measure on $S^1$.

    Indeed, let $A \in \calF$ be a $T$-invariant measurable set and $f$ be the characteristic function of $A$. 
    Then $f \circ R_s = f$ a.e. .
    Expand $f$ into Fourier series, we have 
    \[ f(t) = \sum_{n=-\infty}^\infty c_n e^{2\pi i n t}. \]
    Then we have
    \[ f(t+s) = \sum_{n=-\infty}^\infty c_n e^{2\pi i n (t+s)} = \sum_{n=-\infty}^\infty c_n e^{2\pi i n s} e^{2\pi i n t}. \]
    Hence we have $c_n e^{2\pi i n s} = c_n$ for every $n \in \bbZ$. 
    Since $s$ is irrational, we have $e^{2\pi i n s} \neq 1$ for every $n \neq 0$. 
    Hence $c_n = 0$ for every $n \neq 0$ and $f$ is constant a.e. .
  \end{example}

  \begin{example}
    Let $T: S^1 \to S^1, z \mapsto z^d$ be the power map. 
    Then $T$ is ergodic with respect to the Lebesgue measure on $S^1$.

    Similarly, let $A \in \calF$ be a $T$-invariant measurable set and $f$ be the characteristic function of $A$. 
    Then we have 
    \[ f(t) = \sum_{n=-\infty}^\infty c_n e^{2\pi i n t} = f(d t) = \sum_{n=-\infty}^\infty c_n e^{2\pi i n d t}. \]
    Comparing the Fourier coefficients, we have $c_n = 0$ for every $0 < |n| < d$.
    Doing this repeatedly, we have $c_n = 0$ for every $n \neq 0$ and $f$ is constant a.e. .
  \end{example}

  \begin{theorem}[Birkhoff's Ergodic Theorem]
    Let $T: X \to X$ be a measure-preserving transformation and $f \in L^1(X, \calF, \mu)$. 
    Then for $\mu$-almost every $x \in X$, we have 
    \[ \lim_{N \to \infty} \frac{1}{N} \sum_{n=0}^{N-1} f(T^n x) = \bbE(f | \calI)(x). \]
    In particular, if $T$ is ergodic, then for $\mu$-almost every $x \in X$, we have
    \[ \lim_{N \to \infty} \frac{1}{N} \sum_{n=0}^{N-1} f(T^n x) = \int_X f \, d\mu. \]
  \end{theorem}
  \begin{slogan}
    The time average equals the space average.
  \end{slogan}
  \begin{proof}
    Let $g = f - f_\calI - \varepsilon$ for some $\varepsilon > 0$.
    Note that $g_\calI = f_\calI - f_\calI - \varepsilon = -\varepsilon < 0$ a.e. .
    Set 
    \[ G_n := \max_{0 \leq k \leq n} \ \sum_{i=0}^{k-1} g \circ T^i. \]
    We have $G_{n+1} = \max\{g, g + G_n \circ T\}$ and hence
    \[ g_n := G_{n+1} - G_n \circ T = g - \min\{0, G_n \circ T\} \]
    is monotonically increasing in $n$ for every fixed $x \in X$.
    Set 
    \[ A = \{x \in X : \sup_n G_n(x) \geq 0\}. \]
    Then $A$ is a measurable set and $g_n \to f$ monotonically a.e. on $A$ as $n \to \infty$.
    Then by the monotone convergence theorem, we have
    \[ 0 \leq \int_A G_{n+1} - G_n \, d\mu = \int_A G_{n+1} - G_n \circ T \, d\mu = \int_A g_n \, d\mu \to \int_A g \, d\mu. \]
    Note that $A$ is $T$-invariant.
    Then 
    \[ \int_A g \, d\mu = \int_A g_\calI \, d\mu = -\varepsilon \mu(A) \leq 0. \]
    Hence $\mu(A) = 0$.
    Then we have $G_n \leq 0 \text{ a.e. on } X$ and then 
    \[ \limsup_{n \to \infty} \frac{1}{n} \sum_{i=0}^{n-1} g \circ T^i \leq \limsup_{n \to \infty} \frac{1}{n} G_n \leq 0 \]
    a.e. on $X$.
    Hence we get 
    \[ \limsup_{n \to \infty} \frac{1}{n} \sum_{i=0}^{n-1} f \circ T^i \leq f_\calI + \varepsilon \]
    a.e. on $X$.
    Replace $f$ by $-f$, we get
    \[ \liminf_{n \to \infty} \frac{1}{n} \sum_{i=0}^{n-1} f \circ T^i \geq f_\calI - \varepsilon \]
    a.e. on $X$.
    Hence we have
    \[ \lim_{n \to \infty} \frac{1}{n} \sum_{i=0}^{n-1} f \circ T^i = f_\calI \]
    a.e. on $X$.

    If $T$ is ergodic, then $f_\calI$ is constant a.e. and hence $f_\calI = \int_X f \, d\mu$ a.e. .
    We are done.
      % \Yang{}
  \end{proof}

  \begin{remark}
    Birkhoff's Ergodic Theorem will imply the convergence of in $L^1(X, \calF, \mu)$ as well. 

    If $f$ is bounded, then the assertion is clear by the dominated convergence theorem.
    For general $f \in L^1(X, \calF, \mu)$, we can choose $g \in L^\infty(X, \calF, \mu)$ such that $\|f-g\|_1 < \varepsilon$ for some $\varepsilon > 0$.
    Set $S_n f = \frac{1}{n} \sum_{i=0}^{n-1} f \circ T^i$ for every $n \geq 1$.
    Note that $\|S_n f\|_1 \leq \|f\|_1$ for every $n \geq 1$.
    Then we have 
    \[ \|S_n f - S_m f\|_1 \leq \|S_n g - S_m g\|_1 + \|S_n (f-g)\|_1 + \|S_m (f-g)\|_1 \leq \|S_n g - S_m g\|_1 + 2\varepsilon. \]
    Since $S_n g$ converges in $L^1(X, \calF, \mu)$, we have $\|S_n g - S_m g\|_1 \leq \varepsilon$ for every sufficiently large $n, m$.
    Hence $\{S_n f\}_{n=1}^\infty$ is a Cauchy sequence in $L^1(X, \calF, \mu)$ and hence converges in $L^1(X, \calF, \mu)$.
    % \Yang{}
  \end{remark}


  Applying Birkhoff's Ergodic Theorem to $T: [0,1) \to [0,1), z \mapsto z + \alpha \mod 1$, we obtain the following result.

  \begin{corollary}[Weyl's Equidistribution Theorem]
    Let $\alpha$ be an irrational number. 
    Then the sequence $(n \alpha \mod 1)_{n \in \mathbb{N}}$ is equi-distributed in $[0,1)$.
    More precisely, denote by $\alpha_n = n \alpha \mod 1$, we have 
    \[ \lim_{N \to \infty} \frac{1}{N} \sum_{n=0}^{N-1} \delta_{\alpha_n} = \mu, \]
    where $\delta_x$ is the Dirac measure at $x$ and $\mu$ is the Lebesgue measure on $[0,1)$.
    That is, for every function $f \in C([0,1])$, we have
    \[ \lim_{N \to \infty} \frac{1}{N} \sum_{n=0}^{N-1} f(\alpha_n) = \int_0^1 f(x) \, dx. \]
  \end{corollary}


  \begin{corollary}
    Let $X$ be a topological space and $\calF$ a $\sigma$-algebra on $X$ such that $T: X \to X$ is a measurable transformation.
    Let $\mu_1$ and $\mu_2$ be two different $T$-invariant probability measures on $(X, \calF)$.
    If $T$ is ergodic with respect to $\mu_i$ for $i=1,2$, then $\mu_1$ and $\mu_2$ are mutually singular.
    That is, there exists a measurable set $A$ such that $\mu_1(A) = 1$ and $\mu_2(A) = 0$.
  \end{corollary}
  \begin{proof}
    Since $\mu_1$ and $\mu_2$ are different, there exists a measurable set $B$ such that $\mu_1(B) \neq \mu_2(B)$.
    Apply Birkhoff's Ergodic Theorem to the characteristic function of $B$ and the measure $\mu_i$, there exists a measurable set $A_i$ such that $\mu_i(A_i) = 1$ and for every $x \in A_i$, we have
    \[ \lim_{N \to \infty} \frac{1}{N} \sum_{n=0}^{N-1} \bbone_B(T^n x) = \mu_i(B). \]
    Then we have $A_1 \cap A_2 = \emptyset$ since $\mu_1(B) \neq \mu_2(B)$.
    % \Yang{}
  \end{proof}

  Let $M(\calF, T)$ be the set of all $T$-invariant probability measures on $(X, \calF)$.
  Note that $M(\calF, T)$ is a convex set, that is, if $\mu_1, \mu_2 \in M(\calF, T)$ and $0 < t < 1$, then $t \mu_1 + (1-t) \mu_2 \in M(\calF, T)$.

  \begin{theorem}
    The ergodic measure in $M(\calF,T)$ are exactly the extreme points of $M(\calF,T)$.
    % \Yang{}
  \end{theorem}
  \begin{proof}
    Let $\mu$ be an ergodic measure in $M(\calF,T)$ and $\mu = t \mu_1 + (1-t) \mu_2$ for some $\mu_1, \mu_2 \in M(\calF,T)$ and $0 < t < 1$. 
    If $\mu_1$ is not ergodic with respect to $T$, then there exists a measurable set $A$ such that $\mu_1(A) \in (0,1)$ and $T^{-1}(A) \aeeq A$. 
    Then we have $\mu(A) = t \mu_1(A) + (1-t) \mu_2(A) \in (0,1)$ and $T^{-1}(A) \aeeq A$, which contradicts the ergodicity of $\mu$.
    Hence $\mu_1$ is ergodic with respect to $T$.
    By above corollary, we have $\mu_1$ and $\mu$ must be the same measure since they are not mutually singular.
    Similarly, we have $\mu_2 = \mu$.
    Hence $\mu$ is an extreme point of $M(\calF,T)$.

    Conversely, let $\mu$ be an extreme point of $M(\calF,T)$ and $A$ be a measurable set such that $T^{-1}(A) \aeeq A$. 
    If $t = \mu(A) \in (0,1)$, then we have $\mu = t \mu(-|A) + (1-t) \mu(-|X \setminus A)$, where $\mu(-|A)$ and $\mu(-|X \setminus A)$ are the conditional measures of $\mu$ given $A$ and $X \setminus A$, respectively.
    Note that $\mu(-|A)$ and $\mu(-|X \setminus A)$ are different from $\mu$.
    Hence $\mu$ is not an extreme point of $M(\calF,T)$, which contradicts the assumption.
    Therefore, we have $\mu(A) = 0$ or $\mu(A) = 1$.
    We are done.
    % \Yang{}
  \end{proof}


\subsection{Entropy}

  A \emph{partition} of $X$ is a collection $\calA = \{A_i\}_{i \in I}$ of disjoint measurable subsets of $X$ such that $\bigcup_{i \in I} A_i = X$. 
  For two partitions $\calA$ and $\calB$, the \emph{join} of $\calA$ and $\calB$ is defined as
  \[
  \calA \vee \calB = \{A \cap B : A \in \calA, B \in \calB\}.
  \]
  We say that $\calA$ is \emph{finer} than $\calB$ if for every $A \in \calA$, there exists $B \in \calB$ such that $A \subseteq B$, denoted by $\calA \geq \calB$.
  This is equivalent to saying that $\calA \vee \calB = \calA$.
  For every $x \in X$, we denote by $\calA(x)=A_x$ the unique element of $\calA$ containing $x$.


  \begin{definition}[Information function]
    Let $\calA = \{A_i\}_{i \in I}$ be a countable partition of $X$. The \emph{information function} of $\calA$ is defined as
    \[
    I_\calA(x) = -\log \mu(A_x),
    \]
    where $A_x$ is the unique element of $\calA$ containing $x$.
    % \Yang{}
  \end{definition}


  \begin{definition}[Entropy of a partition]
    Let $\calA = \{A_i\}_{i \in I}$ be a countable partition of $X$. The \emph{entropy} of $\calA$ is defined as
    \[
    H(\calA) = -\sum_{i \in I} \mu(A_i) \log \mu(A_i),
    \]
    where we adopt the convention that $0 \log 0 = 0$.
  \end{definition}

  The information function and the entropy of a partition are related by the following formula:
  \[ H(\calA) = \int_X I_\calA(x) \, d\mu(x). \]

  \begin{example}
    Consider the partition $\calA$ of $[0,1)$ into $n$ equal subintervals, i.e., 
    \[ \calA = \{[0, \frac{1}{n}), [\frac{1}{n}, \frac{2}{n}), \ldots, [\frac{n-1}{n}, 1)\}. \]
    Then the information function $I_{\calA}(x) = \log n$ for every $x \in [0,1)$, and hence the entropy $H(\calA) = \log n$.
    % Then $H(\calA) = \log n$.

    More generally, for $r_1+\cdots + r_k = 1$ with $r_i \geq 0$, consider the partition $\calA = \{A_1, \ldots, A_k\}$ of $[0,1)$ with $A_i = [r_1 + \cdots + r_{i-1}, r_1 + \cdots + r_i)$ for $i=1, \ldots, k$. 
    Then the information function $I_{\calA}(x) = -\log r_i$ for every $x \in A_i$, and hence the entropy $H(\calA) = -\sum_{i=1}^k r_i \log r_i$.
     % \Yang{}
    % \Yang{}
  \end{example}

  \begin{definition}[Conditional entropy and conditional information function]
    Let $\calA$ and $\calB$ be two countable partitions of $X$. 
    The \emph{conditional entropy} of $\calA$ given $\calB$ is defined as
    \[
      H(\calA | \calB) = \sum_{B \in \calB} \mu(B) H(\calA | B),
    \]
    where
    \[
      H(\calA | B) = -\sum_{A \in \calA} \mu(A|B) \log \mu(A|B) = -\sum_{A \in \calA} \frac{\mu(A \cap B)}{\mu(B)} \log \frac{\mu(A \cap B)}{\mu(B)}.
    \]
    The \emph{conditional information function} of $\calA$ given $\calB$ is defined as
    \[
      I_{\calA | \calB}(x) = -\log \mu(A_x | B_x) =  -\log \frac{\mu(A_x \cap B_x)}{\mu(B_x)},
    \]
    where $A_x$ and $B_x$ are the unique elements of $\calA$ and $\calB$ containing $x$, respectively.
    % \Yang{}
  \end{definition}

  We have 
  \[ H(\calA | \{X\}) = H(\calA). \]

  Since 
  \[ \mu(A_x | B_x) = \bbE(\bbone_{A_x} | \sigma(\calB))(x), \]
  the conditional information function is related to the conditional expectation by
  \[ I_{\calA | \calB}(x) = -\log \bbE(\bbone_{A_x} | \sigma(\calB))(x). \]

  % \begin{example}
  %   \Yang{}
  % \end{example}

  Here we list some properties of the entropy and the conditional entropy.

  \begin{proposition}
    Let $\calA, \calB, \calC$ be three countable partitions of $X$. 
    \begin{enumerate}
      \item $H(\calA \vee \calB | \calC) = H(\calA | \calC) + H(\calB | \calA \vee \calC)$. 
      \item $H(\calA \vee \calB) \leq H(\calA) + H(\calB)$.
        % In particular, $H(\calA \vee \calB) = H(\calA) + H(\calB | \calA)$.
      \item $ \calA \leq \calB \implies H(\calA | \calC) \leq H(\calB | \calC)$.
      \item $ \calB \leq \calC \implies H(\calA | \calB) \geq H(\calA | \calC)$.
      \item $H(\calA \vee \calB | \calC) \leq H(\calA | \calC) + H(\calB | \calC)$.
      \item $H(\calA | \calC) \leq H(\calA | \calB) + H(\calB | \calC)$.
    \end{enumerate}
  \end{proposition}
  \begin{proof}
    For (a) and (b). 
    Note that we have 
    \begin{align*}
      & H(\calB| \calA \vee \calC) = \sum_{A \in \calA, C \in \calC} \mu(A \cap C) H(\calB | A \cap C) \\
      =& \sum_{C \in \calC} \mu(C) \sum_{A \in \calA} \mu(A \cap C|C) H(\calB | A \cap C) = \sum_{C \in \calC} \mu(C) H_{\mu(-|C)}(\calB|_C|\calA|_C).
    \end{align*}
    By restricting to each element of $\calC$, we can assume that $\calC = \{X\}$.
    For every $A \in \calA$, we have
    \begin{align*}
      & - \mu(A) \log \mu(A) - \mu(A) \sum_{B \in \calB} \frac{\mu(A \cap B)}{\mu(A)} \log \frac{\mu(A \cap B)}{\mu(A)} \\
      = & - \sum_{B \in \calB} \mu(A \cap B) \log \mu(A) - \sum_{B \in \calB} \mu(A \cap B) \log \frac{\mu(A \cap B)}{\mu(A)} \\
      = & - \sum_{B \in \calB} \mu(A \cap B) \log \mu(A \cap B).
      % = & - \sum_{B \in \calB}\mu(A \cap B) \log \mu(A \cap B).
    \end{align*} 
    Take sum over $A \in \calA$, we get
    \[ H(\calA) + H(\calB | \calA) = H(\calA \vee \calB). \]
    
    For (c).
    By restricting to each element of $\calC$, we can assume that $\calC = \{X\}$.
    For every $A \in \calA$, we have 
    \[ - \sum_{B \in \calB, B \subset A} \mu(B) \log \mu(B) \geq -\sum_{B \in \calB, B \subset A} \mu(B) \log \mu(A) = -\mu(A) \log \mu(A). \]
    Take sum over $A \in \calA$, we get $H(\calB) \geq H(\calA)$.

    For (d).
    First assume $\calB = \{X\}$.
    Since $\phi: x \mapsto -x \log x$ is a concave function on $[0,1]$, we have
    \[ -\mu(A)\log \mu(A) = \phi(\mu(A)) \geq \sum_{C \in \calC} \mu(C) \phi\left(\frac{\mu(A \cap C)}{\mu(C)}\right) = \sum_{C \in \calC} \mu(C) \frac{\mu(A \cap C)}{\mu(C)} \log \frac{\mu(A \cap C)}{\mu(C)}. \]
    Take sum over $A \in \calA$, we get $H(\calA) \geq H(\calA | \calC)$.
    For the general case, we have
    \begin{align*}
      H(\calA | \calB) & = \sum_{B \in \calB} \mu(B) H(\calA | B) \\
      & \geq \sum_{B \in \calB} \mu(B) H_{\mu(-,B)}(\calA |_B, \calC|_B) \\
      & = \sum_{B \in \calB} \mu(B) \sum_{C \in \calC, C \subset B} \mu(C|B) H_{\mu(-,B)}(\calA|_B | C) \\
      & = \sum_{C \in \calC} \mu(C) H(\calA | C) = H(\calA | \calC).
    \end{align*}

    (e) follows from (a) and (d).

    For (f).

    Take integral, we get $H(\calA | \calB) + H(\calB | \calC) \geq H(\calA | \calC)$.

  Let $T: X \to X$ be a measure-preserving transformation. 
  For a partition $\calA$ of $X$, we can define the partition $T^{-1} \calA$ as
  \[ T^{-1} \calA = \{T^{-1} A : A \in \calA\} \]
  and the partition $\calA^n_m$ as
  \[ \calA^n_m = \bigvee_{i=m}^{n-1} T^{-i} \calA. \]
  For simplicity, we write $\calA^n = \calA^n_0$.

  \begin{lemma}
    The sequence $\frac{1}{n+1} H(\calA^n)$ converges as $n \to \infty$.
     % \Yang{}
  \end{lemma}
  \begin{proof}
    Set $f_n = I_{\calA | \calA^n_1}$ for $n \geq 1$ and $f_0 = I_{\calA}$.
    We have 
    \[ I_{\calA^n} = f_n + f_{n-1} \circ T + \cdots + f_1 \circ T^{n-1} + f_0 \circ T^n. \]
    Then we have
    \[ f_n(x) = -\log \mu(\calA(x) | (\calA^n_1)(x)) = -\log \frac{\mu((\calA^n)(x))}{\mu((\calA^{n}_1)(x))} = I_{\calA^n}(x) - I_{\calA^n_1}(x). \]
    Hence we have 
    \begin{align*}
      & f_n + f_{n-1} \circ T + \cdots + f_1 \circ T^{n-1} + f_0 \circ T^n \\
      = & I_{\calA^n}(x) - I_{\calA^n_1}(x) + I_{\calA^{n-1}}(T x) - I_{\calA^{n-1}_1}(T x) + \cdots + I_{\calA^1}(T^{n-1} x) - I_{\calA^1_1}(T^{n-1} x) + I_{\calA}(T^n x) \\
      = & I_{\calA^n}(x) - I_{\calA^n_1}(x) + I_{\calA^{n}_1}(x) - I_{\calA^{n}_2}(x) + \cdots + I_{\calA^n_{n-1}}(x) - I_{\calA^n_n}(x) + I_{\calA^n_n}(x) \\
      = & I_{\calA^n}(x).
    \end{align*}
    Integrating both sides, we get
    \[ H(\calA^n) = H(\calA) + \sum_{j=1}^{n} H(\calA | \calA^j_1). \]
    Here we use the fact that $\int_X f \circ T d \mu = \int_X f \, d\mu$ for every $f \in L^1(X, \calF, \mu)$ since $T$ is measure-preserving.
    
    Note that $H(\calA | \calA^j_1) \leq H(\calA|\calA^k_1)$ for every $j \geq k$ and $H(\calA|\calA^j_1) \leq H(\calA)$ for every $j \geq 1$.
    Hence we have
    \[ n H(\calA^n) = n H(\calA) + n \sum_{j=1}^n H(\calA | \calA^j_1) \leq (n+1) H(\calA) + (n+1) \sum_{k=1}^{n-1} H(\calA | \calA^k_1) = (n+1) H(\calA^{n-1}). \]
    That is, $\frac{1}{n+1} H(\calA^n)$ is a non-increasing sequence.
    Note that $\frac{1}{n+1} H(\calA^n) \geq 0$ for every $n \geq 0$.
    Hence $\frac{1}{n+1} H(\calA^n)$ converges as $n \to \infty$.
  \end{proof}

  \begin{definition}[Entropy of an endomorphism]
    The \emph{entropy of $T: X \to X$ with respect to a partition $\calA$} is defined as
    \[ h_\mu(T, \calA) = \lim_{n \to \infty} \frac{1}{n+1} H(\calA^n). \]
    % where $\calA^n = \bigvee_{i=0}^{n-1} T^{-i} \calA$.
    The \emph{(measure-theoretic) entropy of $T$} is defined as
    \[ h_\mu(T) = \sup_{\calA} h_\mu(T, \calA), \]
    where the supremum is taken over all finite measurable partitions $\calA$ of $X$.    
  \end{definition}

  \begin{example}
    Let $T: S^1 \to S^1, z \mapsto z^2$ be the square map and $\calA = \{[0, \frac{1}{2}), [\frac{1}{2}, 1)\}$ be the partition of $S^1$ into two equal arcs.
    Then we have 
    \[ \calA^n = T^{-n+1} \calA = \{[0, \frac{1}{2^n}), [\frac{1}{2^n}, \frac{2}{2^n}), \ldots, [\frac{2^n-1}{2^n}, 1)\}. \]
    Hence $H(\calA^n) = n \log 2$ and $h_\mu(T, \calA) = \log 2$.
      % \Yang{}
  \end{example}

  \begin{proposition}
    Let $T: X \to X$ be a measure-preserving transformation and $\calA, \calB$ be two countable measurable partitions of $X$ or finite entropy. 
    \begin{enumerate}
      \item $h_\mu(T, \calA) \leq H(\calA)$.
      \item $h_\mu(T, \calA \vee \calB) \leq h_\mu(T, \calA) + H(T,\calB)$.
      \item $\calA \leq \calB \implies h_\mu(T, \calA) \leq h_\mu(T, \calB)$.
      \item $h_\mu(T, \calA) \leq h_\mu(T, \calB) + H(\calA | \calB)$.
      \item $h(T, T^{-1} \calA) = h(T, \calA)$.
      \item $h(T, \calA^k) = h(T, \calA)$ for every $k \geq 1$.
      \item If $T$ is invertible, then $h(T, \bigvee_{i=-k}^{k} T^i \calA) = h(T, \calA)$ for every $k \geq 1$.
    \end{enumerate}
    % \Yang{}
  \end{proposition}
  \begin{proof}
    (a)(b)(c) follow from the properties of the entropy and the conditional entropy.
    For (d), we have
    \begin{align*}
      H(\calA^n) & = H(\calB^n) + H(\calA^n | \calB^n) \\
        & \leq H(\calB^n) + \sum_{j=0}^{n-1} H(T^{-j} \calA | \calB^n) \\
        & \leq H(\calB^n) + \sum_{j=0}^{n-1} H(T^{-j} \calA | T^{-j} \calB) \\
        & = H(\calB^n) + n H(\calA | \calB).
    \end{align*}
    Divide by $n+1$ and let $n \to \infty$, we get the desired inequality.

    (e) follows from the fact that $H(T^{-1} \calA^n) = H(\calA^n)$ since $T$ is measure-preserving.

    For (f), note that 
    \[ (\calA^k)^n = \bigvee_{i=0}^{n-1} T^{-i} \calA^k = \bigvee_{i=0}^{n-1} \bigvee_{j=0}^{k-1} T^{-i-j} \calA = \bigvee_{m=0}^{n+k-2} T^{-m} \calA = \calA^{n+k-1}. \]
    Similar for (g).
    % since $\calB = \calA^k$ is finer than $\calA$, we have $h(T, \calA^k) \geq h(T, \calA)$.
    % \Yang{}
  \end{proof}

  \begin{proposition}
    Let $T: X \to X$ be a measure-preserving transformation.
    Then 
    \[ h(T^k) = k h(T) \]
    for every positive integer $k$.
    Furthermore, if $T$ is invertible, then 
    \[ h(T^{-1}) = h(T). \]
      % \Yang{}
  \end{proposition}
  \begin{proof}
    Given a finite measurable partition $\calA$ of $X$, set $\calB = \bigvee_{j=0}^{k-1} T^{-j} \calA$.
    we have
    \[ \calA^{nk} = \bigvee_{i=0}^{nk-1} T^{-i} \calA = \bigvee_{m=0}^{n-1} \ T^{-m k} \bigvee_{j=0}^{k-1}  T^{-j} \calA = \bigvee_{m=0}^{n-1} T^{-m k} \calB = \calB^n. \]
    Then 
    \[ h(T, \calA) = \lim_{n \to \infty} \frac{1}{nk} H(\calA^{nk}) = \lim_{n \to \infty} \frac{1}{nk} H(\calB^n) = \frac{1}{k} h(T^k, \calB) \leq \frac{1}{k} h(T^k). \]
    Take supremum over $\calA$, we get $h(T) \leq \frac{1}{k} h(T^k)$.
    On the other hand, we have 
    \[ H(\calA^n(T^k)) = H\left(\bigvee_{i=0}^{n-1} T^{-i k} \calA\right) \leq H\left(\bigvee_{i=0}^{n-1} \bigvee_{j=0}^{k-1} T^{-i k - j} \calA\right) = H(\calA^{nk}). \]
    Hence we have
    \[ h(T^k, \calA) = \lim_{n \to \infty} \frac{1}{n} H(\calA^n(T^k)) \leq \lim_{n \to \infty} \frac{1}{n} H(\calA^{nk}) = k h(T, \calA). \]
    Take supremum over $\calA$, we get $h(T^k) \leq k h(T)$.
    Hence $h(T^k) = k h(T)$.

    If $T$ is invertible, note that
    \[ H(\calA^n(T^{-1})) = H\left(\bigvee_{i=0}^{n-1} T^{i} \calA\right) \leq H\left(T^{n-1} \left(\bigvee_{i=0}^{n-1} T^{-i} \calA\right) = H\left(\bigvee_{i=0}^{n-1} T^{-i} \calA\right) = H(\calA^n(T))\right). \]
    Hence we have $h(T^{-1}, \calA) = h(T, \calA)$ for every finite measurable partition $\calA$ of $X$.
    % \Yang{}
  \end{proof}

  \begin{theorem}[Shannon-McMillan-Breiman Theorem]
    Let $T: X \to X$ be an ergodic measure-preserving transformation and $\calA$ be a countable measurable partition of $X$ with finite entropy. 
    Then for $\mu$-almost every $x \in X$, we have
    \[
      \lim_{n \to \infty} \frac{1}{n+1} I_{\calA^n}(x) = h_\mu(T, \calA).
    \]
  \end{theorem}
  \begin{slogan}
    The ``local entropy'' $\lim_{n \to \infty} \frac{1}{n+1} I_{\calA^n}(x)$ equals the ``global entropy'' $h(T, \calA)$ for a.e. $x$.
  \end{slogan}
  \begin{proof}
    \hspace{1em}

    \begin{step}
      Set $f_n = I_{\calA | \calA^n_1}$ for $n \geq 1$ and $f_0 = I_{\calA}$.
      We have 
      \[ I_{\calA^n} = f_n + f_{n-1} \circ T + \cdots + f_1 \circ T^{n-1} + f_0 \circ T^n. \]
    \end{step}

    By the same calculation as in the proof of the previous lemma.

    We have 
    \[ f_n(x) = -\log \mu(\calA(x) | (\calA^n_1)(x)) = -\log \frac{\mu((\calA^n)(x))}{\mu((\calA^{n}_1)(x))} = I_{\calA^n}(x) - I_{\calA^n_1}(x). \]
    Hence we have 
    \begin{align*}
      & f_n + f_{n-1} \circ T + \cdots + f_1 \circ T^{n-1} + f_0 \circ T^n \\
      = & I_{\calA^n}(x) - I_{\calA^n_1}(x) + I_{\calA^{n-1}}(T x) - I_{\calA^{n-1}_1}(T x) + \cdots + I_{\calA^1}(T^{n-1} x) - I_{\calA^1_1}(T^{n-1} x) + I_{\calA}(T^n x) \\
      = & I_{\calA^n}(x) - I_{\calA^n_1}(x) + I_{\calA^{n}_1}(x) - I_{\calA^{n}_2}(x) + \cdots + I_{\calA^n_{n-1}}(x) - I_{\calA^n_n}(x) + I_{\calA^n_n}(x) \\
      = & I_{\calA^n}(x).
    \end{align*}

    \begin{step}
      The sequence $(f_n)_{n \geq 0}$ converges a.e. and in $L^1(X, \calF, \mu)$.
      Denote the limit by $f$.
    \end{step}
    Note that $A_1^n$ is ascending in $n$.
    By the martingale convergence theorem, we have $\bbE(\bbone_{\calA} | \sigma(\calA^n_1))$ converges a.e. on $A \in \calA$.
    Hence $f_n|_A = -\log \bbE(\bbone_{\calA} | \sigma(\calA^n_1))$ converges a.e. on $A$.
    Then $f_n$ converges a.e. on $X$.
    Set $f^* = \sup_{n \geq 0} f_n$.
    By dominated convergence theorem, we only need to show that $f^* \in L^1(X, \calF, \mu)$.

    For every $A \in \calF$, set 
    \[ A' = \{x \in A : f^*(x) > \lambda\} \]
    for $\lambda > 0$.
    Then I claim that $\mu(A') \leq e^{-\lambda}$.
    Set $f_n^A = -\log \bbE(\bbone_A | \sigma(\calA^n_1))$ for $n \geq 0$ and 
    \[ B_n^A = \{x \in A : f_n^A(x) > \lambda, f_j^A(x) \leq \lambda \text{ for every } j < n\} \]
    for $n \geq 0$.
    Then we have $A' = \bigsqcup_{n=0}^{\infty} (B_n^A \cap A)$ and $B_n^A \in \sigma(\calA^n_1)$ for every $n \geq 0$.
    Hence we have
    \[ \mu(B_n^A \cap A) = \int_{B_n^A} \bbE(\bbone_A | \sigma(\calA^n_1)) \, d\mu = \int_{B_n^A} e^{-f_n^A} \, d\mu \leq e^{-\lambda} \mu(B_n^A). \]
    Then 
    \[ \mu(A') = \sum_{n=0}^{\infty} \mu(B_n^A \cap A) \leq e^{-\lambda} \sum_{n=0}^{\infty} \mu(B_n^A) \leq e^{-\lambda}. \]
    Hence we have
    \begin{align*}
      \int_X f^* \, d\mu & = \sum_{A \in \calA} \int_A f^* \, d\mu = \sum_{A \in \calA} \int_0^{\infty} \mu(\{x \in A : f^*(x) > \lambda\}) \, d\lambda \\
      & \leq \sum_{A \in \calA} \int_0^{-\log \mu(A)} \mu(A) \, d\lambda + \sum_{A \in \calA} \int_{-\log \mu(A)}^{\infty} e^{-\lambda} \, d\lambda \\
      & = \sum_{A \in \calA} \mu(A) (-\log \mu(A)) + \sum_{A \in \calA} e^{-\lambda} \bigg|_{-\log \mu(A)}^{\infty} \\
      &= H(\calA) + \sum_{A \in \calA} \mu(A) = H(\calA) + 1 < \infty.
    \end{align*}


    \begin{step}
      $f_\calI = \lim_{n \to \infty} \frac{1}{n+1} I_{\calA^n}$ a.e. and in $L^1(X, \calF, \mu)$.
    \end{step}

    We have
    \[ \left| \frac{1}{n+1} I_{\calA^n} - f_\calI\right| \leq \frac{1}{n+1} \sum_{i=0}^{n} \left|f_{n-i} \circ T^i - f \circ T^i\right| + \left|\frac{1}{n+1} \sum_{i=0}^{n} f \circ T^i - f_\calI\right|. \]
    By Birkhoff's ergodic theorem, the last term converges to $0$ a.e. and in $L^1(X, \calF, \mu)$.
    Set $g_n = |f-f_n|$ for $n \geq 0$ and $G_N = \sup_{n \geq N} g_n$ for $N \geq 0$.
    Then we have $g_n \to 0$ a.e. and in $L^1(X, \calF, \mu)$ as $n \to \infty$.
    And $G_0 \leq f^* + f \in L^1(X, \calF, \mu)$ and $G_N$ is non-increasing in $N$.
    Hence we have $G_N \to 0$ a.e. monotonically as $N \to \infty$.
    We have 
    \begin{align*}
      \frac{1}{n+1} \sum_{i=0}^{n} g_{n-i} \circ T^i & = \frac{1}{n+1} \sum_{i=0}^{n - N-1} g_{n-i} \circ T^i + \frac{1}{n+1} \sum_{i=n-N}^{n} g_{n-i} \circ T^i \\
      & \leq \frac{1}{n+1} \sum_{i=0}^{n - N-1} G_N \circ T^i + \frac{1}{n+1} \sum_{i=n-N}^{n} G_0 \circ T^i.
    \end{align*}
    Take limit as $n \to \infty$, we have
    \[ \lim_{n \to \infty} \frac{1}{n+1} \sum_{i=0}^{n} g_{n-i} \circ T^i \leq (G_N)_\calI \]
    a.e. by Birkhoff's ergodic theorem.
    Note that $(G_N)_\calI$ is non-increasing in $N$, $(G_N)_\calI \geq 0$ and 
    \[ \int_X (G_N)_\calI \, d\mu = \int_X G_N \, d\mu \to 0 \]
    as $N \to \infty$.
    Hence $(G_N)_\calI \to 0$ a.e. as $N \to \infty$.
    Then we have 
    \[ \lim_{n \to \infty} \frac{1}{n+1} \sum_{i=0}^{n} g_{n-i} \circ T^i = 0 \text{ a.e.} \]
    The convergence in $L^1(X, \calF, \mu)$ is easy since 
    \[ \int \frac{1}{n+1} \sum_{i=0}^{n} g_{n-i} \circ T^i \, d\mu = \frac{1}{n+1} \sum_{i=0}^{n} \int g_{n-i} \, d\mu, \]
    and the right-hand side converges to $0$ in $L^1(X, \calF, \mu)$ as $n \to \infty$ since $g_n \to 0$ in $L^1(X, \calF, \mu)$ as $n \to \infty$.


    % \Yang{}

    \begin{step}
      Finish the proof.
    \end{step}
    We have 
    \begin{align*}
      h(T, \calA) & = \lim_{n \to \infty} \frac{1}{n+1} H(\calA^n) = \lim_{n \to \infty} \frac{1}{n+1} \int_X I_{\calA^n}(x) \, d\mu(x) \\
      & = \lim_{n \to \infty} \frac{1}{n+1} \sum_{i=0}^{n} \int_X f_i \circ T^i \, d\mu(x) = \lim_{n \to \infty} \frac{1}{n+1} \sum_{i=0}^{n} \int_X f_i \, d\mu(x)\\
      & = \int_X f \, d\mu(x).
    \end{align*}
    % \[ h(T, \calA) = \lim_{n \to \infty} \frac{1}{n+1} H(\calA^n) = \lim_{n \to \infty} \frac{1}{n+1} \int_X I_{\calA^n}(x) \, d\mu(x) = \lim_{n \to \infty} \frac{1}{n+1} \sum_{i=0}^{n} \int_X f_i \circ T^i \, d\mu(x) = \lim_{n \to \infty} \frac{1}{n+1} \sum_{i=0}^{n} \int_X f_i \, d\mu(x) = \int_X f \, d\mu(x). \]
    Since $T$ is ergodic, we have $f_\calI = \int_X f \, d\mu$ a.e.
    % \Yang{}
  \end{proof}
