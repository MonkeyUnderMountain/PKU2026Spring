  A morphism $\calX \to \calY$ of categories over $\Sch$ is representable by schemes if 
  $\forall T \to \calY$ from a scheme $T$, $\calX \times_\calY T$ is (represented by) a scheme.

  If $\calP$ is a property of morphisms of schemes, stable under base change, 
  then a morphism $\calf: \calX \to \calY$ representable by schemes has property $\calP$ if 
  $\forall T \to \calY$ from a scheme $\calX \times_\calY T \to T$ has property $\calP$.

  An algebraic space is a sheaf $\calX$ over $\Sch_{\et}$ s.t. $\exists U \in \Sch$ and $U \to \calX$ surjective, \'etale and representable by schemes.

  Then we have a further definition of representable morphism.
  A morphism $\calX \to \calY$ of categories over $\Sch$ is \emph{representable} if 
  $\forall T \to \calY$ from a scheme $T$, $\calX \times_\calY T$ is an algebraic space.

  If $\calP$ is a property of morphisms of schemes, stable under base change and \'etale local on the source,
  then a representable morphism $\calf: \calX \to \calY$ has property $\calP$ if 
  $\forall T \to \calY$ from a scheme, denote by $U \to \calX \times_\calY T$ a \'etale presentation of $\calX \times_\calY T$, then $U \to \calX \times_\calY T \to T$ has property $\calP$.

  \begin{definition}
    A stack $\calX/S$ over $\Sch_\et$ is an \emph{algebraic stack} (resp. \emph{Deligne-Mumford (DM) stack}) if $\exists U \in \Sch$, surjective smooth (resp. \'etale) representable morphism.

    A morphism of algebraic (resp. DM) stacks is a morphism of categories.
  \end{definition}

  If $\calX_1 \to \calX_3$, $\calX_2 \to \calX_3$ morphism of algebraic (resp. DM) stacks, then $\calX_1 \times_{\calX_3} \calX_2$ is an algebraic (resp. DM) stack.

\subsection[Morphism between moduli of stable curves]{Morphisms between $\overline{\calM}_{g,n}$}

  To understand the geometry of $\overline{\calM}_{g,n}$, we first try to give a stratification of $\overline{\calM}_{g,n}$ by topological type of curves, which is encoded by some combinatorial data called stable graph.
  For every irreducible component of a stable curve, we associate a vertex, and for every node, we associate an edge connecting the two vertices corresponding to the two components intersecting at the node.
  And then label these by some numerical data, such as the genus of each component, the number of marked points on each component, etc.
  We get the following definition of stable graph.

  \begin{definition}
    A \emph{stable graph} is a tuple $\Gamma = (V,H,L,g,a,i,\zeta)$, where
    \begin{itemize}
      \item $V=V(\Gamma)$ is nonempty finite set (vertices of $\Gamma$);
      \item $H= H (\Gamma)$ is a finite set (half-edges of $\Gamma$);
      \item $g: V \to \bbZ_{\geq 0}$ a map;
      \item $a:H \to V$, associating to $h \in H$ a vertex $a(h)$ incident to $V$; 
      \item $H(v) = \{h \in H| a(h) = v\}$;
      \item $i:H \to H$ is an involution (send a half-edge to the other half-edge of the same edge, or send a leg to itself);
      \item $L = L(\Gamma) = \{l \in H | i(l) = l\}$ (legs of $\Gamma$);
      \item $E(\Gamma) = \{e = \{h,h'\} \subset H | h\neq h', i(h) = h'\}$ (edges of $\Gamma$);
      \item $\zeta: L \to \{1,\cdots,n\}$ a bijection (labeling of legs);
    \end{itemize}
    such that the resulted graph is connected and $2g(v) - 2 + n(v) > 0$ for all $v \in V$, where $n(v) = |H(v)|$.

    Set $g(\Gamma) \coloneqq \sum_{v \in V} g(v) + 1 + |E(\Gamma)| - |V|$ be the \emph{genus} of $\Gamma$ and $n(\Gamma) = |L(\Gamma)|$ be the \emph{number of legs} of $\Gamma$.
  \end{definition}

  \begin{definition}\label{def:morphisms_inverse_image_and_G_g_n_of_stable_graphs}
    Let $\Gamma,\Gamma'$ be two stable graphs.
    A morphism $\varphi: \Gamma' \to \Gamma$ of stable graphs is a pair of maps 
    \[ \varphi = \left( \begin{tikzcd}[row sep=1ex]
      V' \arrow[r, "\varphi_V", two heads] & V,\\
      H'  & H \arrow[l, "\varphi_H"', hook']
    \end{tikzcd} \right)
    \]
    (note the \textbf{direction} of the arrows) such that
    \begin{itemize}
      \item $\varphi_H$ is injective, and sends edges (resp. legs) of $\Gamma$ to the edges (resp. legs) of $\Gamma'$;
      \item $\varphi_V$ is surjective with the commutate diagram
        \[ \begin{tikzcd}
          H'  \arrow[d, "a'"] & H \arrow[d, "a"] \arrow[l, "\varphi_H"'] \\
          V' \arrow[r, "\varphi_V"] & V.
        \end{tikzcd} \]
      \item $\forall v \in V$, its preimage 
        \[ \Gamma'_v \coloneqq \varphi_V^{-1}(v) \coloneqq \bigg\{ V(\Gamma'_v) = \varphi_V^{-1}(v),\quad H(\Gamma'_v) = \{h' \in H(\Gamma') | a(h') \in V(\Gamma'_v)\} \bigg\} \]
        is a stable graph with genus $g(\Gamma'_v) = g(v)$ and number of legs $n(\Gamma'_v) = n(v)$.
    \end{itemize}
  \end{definition}

  In the following, $g,n \geq 0$, $2g-2+n > 0$ are fixed.

  Let $(C,p_1,\cdots,p_n)$ be a stable $n$-pointed curve of genus $g$.
  Let $\nu: \tilde{C} = \bigsqcup_{i \in I} \tilde{C_i} \to C$ be the normalization of $C$.
  Let $S \subset C$ the set of nodes of $C$.

  \begin{definition}
    The dual graph $\Gamma_C$ of $C$ is defined as $\Gamma_C = (V,H,L,g,a,i,\zeta)$:
    \begin{itemize}
      \item $V \cong \{\tilde{C_i} | i \in I\}$;
      \item $H = \{\nu^{-1}(s) | s \in S\} \cup \{p_1,\cdots,p_n\}$;
      \item $L = \{p_1,\cdots,p_n\}$;
      \item $g: V \to \bbZ_{\geq 0}$, $\tilde{C_i} \mapsto g(\tilde{C_i})$;
      \item $a: H \to V$, $q' \mapsto \tilde{C_i}$ if $q' \in \tilde{C_i}$;
      \item $i:H \to H$, $i(q') = q''$ if $\nu(q') = \nu(q'')$;
      \item $\zeta: L \to \{1,\cdots,n\}$, $p_j \mapsto j$.
    \end{itemize}
  \end{definition}

  \begin{remark}
    We have an exact sequence of groups
    \[ 1 \to \Aut(\tilde{C},q \in H(\Gamma_C)) \to \Aut(C,p_1,\cdots,p_n) \to \Aut(\Gamma_C) \to 1. \]
  \end{remark}

  \begin{theorem}
    Let $(\calC \to S, \sigma_1,\sigma_2)$ be a family of stable $2$-pointed curves of (possibly disconnected) prestable curves over a scheme $S$ 
    such that $\sigma_1(s)$ and $\sigma_2(s)$ are smooth distinct points of $\calC_s$ for all $s \in S$.
    Then there exists a canonical finite morphism $\xi: \calC \to \calC'$ of schemes over $S$ such that
    \begin{enumerate}[label=(\arabic*)]
      \item $\phi': \calC' \to S$ is a family of prestable curves with a section $\sigma': S \to \calC'$ such that $\sigma' = \xi \circ \sigma_1 = \xi \circ \sigma_2$ and $\sigma'(s)$ is a node of $\calC'_s$ for all $s \in S$;
      \item $\xi$ restricts to an isomorphism $\calC \setminus (\sigma_1(S) \cup \sigma_2(S)) \to \calC' \setminus \sigma'(S)$ and there is an identification of $|\calC|$ with $|\calC'/\sim|$ under the equivalence relation $\sigma_1(s) \sim \sigma_2(s)$ for all $s \in S$;
      \item $\forall U \subset \calC'$ open, $\Gamma(U, \calO_{\calC'}) = \{h \in \Gamma(\xi^{-1}(U), \calO_\calC) | \sigma_1^*h = \sigma_2^*h\}$.
    \end{enumerate}
    This construction is compatible with base change $\calS' \to \calS$.
  \end{theorem}
