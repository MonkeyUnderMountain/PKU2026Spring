\section{March 16: Review of tautological rings}

\begin{definition}
  The \emph{tautological ring} $R^*(\overline{\calM}_{g,n})$ is the smallest $\bbQ$-subalgebra of $A^*(\overline{\calM}_{g,n})$ that contains $1 = [\overline{\calM}_{g,n}]$, and is closed under pushforward by all glueing morphisms \Yang{} and all forgetful morphisms $\pi: \overline{\calM}_{g,n} \to \overline{\calM}_{g,n-1}$.
  A \emph{tautological class} is an element of the tautological ring.
\end{definition}

  Let $\Gamma \in G_{g,n}$, $l = |E(\Gamma)|$.
  Then 
  \[ \delta_\Gamma = \frac{1}{|\Aut(\Gamma)|} \xi_{\Gamma*} [\overline{\calM}_\Gamma] \in A^l(\overline{\calM}_{g,n}). \]
  For $v \in V(\Gamma)$, let $\alpha_v \in A^*(\overline{\calM}_{g(v),n(v)})$, $\pi_v: \overline{\calM}_\Gamma \to \overline{\calM}_{g(v),n(v)}$ be the projection.
  Set $\alpha = \prod_{v \in V(\Gamma)} \pi_v^* \alpha_v$.
  If $\alpha_v$ are tautological classes for all $v \in V(\Gamma)$, then $\xi_{\Gamma*} \alpha$ is a tautological class.
